        function applyTranslations() {

            // 1. Textos Generales (usando data-i18n)

                const key = el.getAttribute('data-i18n');
                if (t[key]) {
                    el.innerHTML = t[key];
                }
            });

            // 2. Textos Espec√≠ficos de Botones de Idioma Contextuales
            document.getElementById('btn-lang-home-text').innerText = t.btn_lang_home;
            document.getElementById('btn-lang-social-text').innerText = t.btn_lang_social;

        function updatePrice(itemId, size) {
            const item = menuData.find(i => i.id === itemId);
            if(!item) return;
            const priceEl = document.getElementById(`price-${itemId}`);
            priceEl.innerText = formatPrice(item.prices[size]);
            const sizes = ['small', 'medium', 'large'];
            sizes.forEach(s => {
                const btn = document.getElementById(`btn-${itemId}-${s}`);
                if(s === size) {
                    btn.classList.add('active');
                    btn.classList.remove('text-gray-400');
                } else {
                    btn.classList.remove('active');
                    btn.classList.add('text-gray-400');
                }
            });

            const menu = document.getElementById('mobile-menu');
            const icon = document.getElementById('menu-icon');
            
            if (menu.classList.contains('hidden')) {
                menu.classList.remove('hidden');
                icon.classList.remove('fa-bars');
                icon.classList.add('fa-times');
            } else {
                menu.classList.add('hidden');
                icon.classList.remove('fa-times');
                icon.classList.add('fa-bars');
            }
        }
        
        function triggerAnimation(elementId) {
            const el = document.getElementById(elementId);
            el.classList.remove('page-transition');
            void el.offsetWidth; 
            el.classList.add('page-transition');
        }

        function showHome() {
            document.getElementById('home-view').classList.remove('hidden-view');
            document.getElementById('menu-view').classList.add('hidden-view');
            document.getElementById('social-view').classList.add('hidden-view');
            triggerAnimation('home-view');
            window.scrollTo(0,0);
        }

        function showMenuPage() {
            document.getElementById('home-view').classList.add('hidden-view');
            document.getElementById('menu-view').classList.remove('hidden-view');
            document.getElementById('social-view').classList.add('hidden-view');
            triggerAnimation('menu-view');
            window.scrollTo(0,0);
        }

        function showSocialPage() {
            document.getElementById('home-view').classList.add('hidden-view');
            document.getElementById('menu-view').classList.add('hidden-view');
            document.getElementById('social-view').classList.remove('hidden-view');
            triggerAnimation('social-view');
            window.scrollTo(0,0);
        }
